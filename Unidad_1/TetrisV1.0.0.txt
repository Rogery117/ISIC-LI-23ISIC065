.model small
.stack 100h

.data
    OFFSET_X     equ 120
    OFFSET_Y     equ 10
    board        db 200 dup(0)
    grid_x       dw 4
    grid_y       dw 0
    tick_count   dw 0
    tick_limit   dw 20     
    game_speed   dw 8000h 

.code
inicio:
    mov ax, @data
    mov ds, ax

    ; Iniciar Modo Gráfico 13h (320x200)
    mov ax, 13h
    int 10h

game_loop:
    call draw_everything
    
    ; Revisar si hay tecla
    mov ah, 01h
    int 16h
    jz gravity_tick
    
    ; Leer tecla
    mov ah, 00h
    int 16h
    
    cmp ah, 4Bh ; Izquierda
    je move_l
    cmp ah, 4Dh ; Derecha
    je move_r
    cmp ah, 50h ; Abajo (Acelerar)
    je fast_drop
    cmp al, 1Bh ; ESC
    je exit_p
    jmp gravity_tick

move_l:
    dec grid_x
    call check_col
    or al, al
    jz gravity_tick
    inc grid_x 
    jmp gravity_tick

move_r:
    inc grid_x
    call check_col
    or al, al
    jz gravity_tick
    dec grid_x 
    jmp gravity_tick

fast_drop:
    mov tick_count, 20 ; Forzamos la caída
    jmp gravity_tick

gravity_tick:
    inc tick_count
    mov ax, tick_count
    cmp ax, tick_limit
    jl wait_f
    
    mov tick_count, 0
    inc grid_y
    call check_col
    or al, al
    jz wait_f 
    
    ; Bloqueo de pieza al chocar
    dec grid_y
    call lock_p
    mov grid_x, 4
    mov grid_y, 0
    
    ; Game Over check
    call check_col
    or al, al
    jnz exit_p

wait_f:
    mov cx, 00h
    mov dx, game_speed
    mov ah, 86h
    int 15h
    jmp game_loop

exit_p:
    mov ax, 03h ; Volver a modo texto
    int 10h
    mov ax, 4C00h
    int 21h

; --- FUNCIONES DE LÓGICA ---

check_col proc
    cmp grid_x, 0
    jl is_c
    cmp grid_x, 9
    jg is_c
    cmp grid_y, 19
    jg is_c
    
    mov ax, grid_y
    mov bl, 10
    mul bl
    add ax, grid_x
    mov si, ax
    cmp board[si], 1
    je is_c
    mov al, 0
    ret
is_c:
    mov al, 1
    ret
check_col endp

lock_p proc
    mov ax, grid_y
    mov bl, 10
    mul bl
    add ax, grid_x
    mov si, ax
    mov board[si], 1
    ret
lock_p endp

; --- FUNCIONES DE GRÁFICOS ---

draw_everything proc
    mov ax, 0A000h
    mov es, ax
    xor di, di
    mov cx, 32000
    xor ax, ax
    rep stosw ; Limpiar pantalla

    ; Dibujar bloques del tablero
    xor si, si
    mov dx, 0 ; Y
dy: mov bx, 0 ; X
dx_l:
    cmp board[si], 1
    jne nxt
    push dx
    push bx
    mov ax, dx ; Fila
    mov cl, 07h ; Color Gris
    call draw_sq
    pop bx
    pop dx
nxt: inc si
    inc bx
    cmp bx, 10
    jl dx_l
    inc dx
    cmp dx, 20
    jl dy

    ; Dibujar bloque actual
    mov ax, grid_y
    mov bx, grid_x
    mov cl, 0Bh ; Color Cian
    call draw_sq
    ret
draw_everything endp

draw_sq proc
    push ax
    push bx
    ; Convertir grid a pixeles
    shl ax, 3 ; y * 8
    add ax, OFFSET_Y
    shl bx, 3 ; x * 8
    add bx, OFFSET_X
    
    ; Calcular dirección en memoria de video
    mov dx, 320
    mul dx
    add ax, bx
    mov di, ax
    
    mov al, cl ; Color
    mov cx, 8
rw: push cx
    mov cx, 8
    rep stosb
    add di, 320-8
    pop cx
    loop rw
    pop bx
    pop ax
    ret
draw_sq endp

end inicio
